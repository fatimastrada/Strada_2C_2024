168static void OperarConDistancia(void *pvParameter){
169 while(true){
170 // La tarea esta en espera (bloqueada) hasta que reciba una notificación mediante ulTaskNotifyTake
171 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
172 if(on == 1)
173 {
174 // Medir distancia
175 distancia = HcSr04ReadDistanceInCentimeters();
176
177 // Manejar LEDs según la distancia
178 if(distancia < 10){
179 LedOff(LED_1);
180 LedOff(LED_2);
181 LedOff(LED_3);
182 }
183 else if (distancia >= 10 && distancia < 20){
184 LedOn(LED_1);
185 LedOff(LED_2);
186 LedOff(LED_3);
187 }
188 else if(distancia >= 20 && distancia < 30){
189 LedOn(LED_1);
190 LedOn(LED_2);
191 LedOff(LED_3);
192 }
193 else if(distancia >= 30){
194 LedOn(LED_1);
195 LedOn(LED_2);
196 LedOn(LED_3);
197 }
198
199 // Mostrar distancia en la pantalla LCD si no está en hold
200 if(hold == 0){
201 LcdItsE0803Write(distancia);
202 }
203 }
204 else
205 {
206 // Apagar LEDs y pantalla LCD si on está apagado
207 LedOff(LED_1);
208 LedOff(LED_2);
209 LedOff(LED_3);
210 LcdItsE0803Off();
211 }
212
213 }
214}

228// Inicializa los periféricos, configura el temporizador y las interrupciones de teclas, y crea las tareas del sistema.
229void app_main(void) {
230
231 LedsInit();
232 HcSr04Init(3, 2);
233 LcdItsE0803Init();
234 SwitchesInit();
235 // timer_config_t: configura el temporizador definiendo su periodo y la función que se llamará cuando el temporizador
236 //se dispare.
237 timer_config_t timer_sensor = {
238 .timer = TIMER_A,
239 .period = CONFIG_SENSOR_TIMERA,
240 .func_p = FuncTimerA,
241 .param_p = NULL
242 };
243
244 serial_config_t serial_port = {
245 .port = UART_PC,
246 .baud_rate = 115200,
247 .func_p = &RecibirData,
248 .param_p = NULL
249 };
250
251 UartInit(&serial_port);
252 TimerInit(&timer_sensor);
253 // Se habilitan interrupciones cuando se presionan las teclas SWITCH_1 y SWITCH_2 (desencadena OperarConTeclado)
254 SwitchActivInt(SWITCH_1, &Interrupciontecla1, NULL );
255 SwitchActivInt(SWITCH_2, &Interrupciontecla2, NULL );
256
257 xTaskCreate(&OperarConDistancia, "OperarConDistancia", 2048, NULL, 5, &operar_distancia_task_handle);
258 xTaskCreate(&EnviarDistancia, "EnviarDist", 512, NULL, 4, &enviar_distancia_task_handle);
259
260 TimerStart(timer_sensor.timer);
261}
