101static void OperarConDistancia(void *pvParameter){
102 while(true){
103 // La tarea esta en espera (bloqueada) hasta que reciba una notificación mediante ulTaskNotifyTake
104 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
105
106 if(on == 1)
107 {
108 // Medir distancia
109 distancia = HcSr04ReadDistanceInCentimeters();
110
111 // Manejar LEDs según la distancia
112 if(distancia < 10){
113 LedOff(LED_1);
114 LedOff(LED_2);
115 LedOff(LED_3);
116 }
117 else if (distancia >= 10 && distancia < 20){
118 LedOn(LED_1);
119 LedOff(LED_2);
120 LedOff(LED_3);
121 }
122 else if(distancia >= 20 && distancia < 30){
123 LedOn(LED_1);
124 LedOn(LED_2);
125 LedOff(LED_3);
126 }
127 else if(distancia >= 30){
128 LedOn(LED_1);
129 LedOn(LED_2);
130 LedOn(LED_3);
131 }
132
133 // Mostrar distancia en la pantalla LCD si no está en hold
134 if(hold == 0){
135 LcdItsE0803Write(distancia);
136 }
137 }
138 else
139 {
140 // Apagar LEDs y pantalla LCD si on está apagado
141 LedOff(LED_1);
142 LedOff(LED_2);
143 LedOff(LED_3);
144 LcdItsE0803Off();
145 }
146
147 }
148}
