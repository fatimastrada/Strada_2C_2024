119int8_t convertToBcdArray(uint32_t data, uint8_t digits, uint8_t *bcd_number) {
120 // Verificamos si el número de dígitos proporcionado es suficiente
121 uint32_t max_value = 1;
122 for (uint8_t i = 0; i < digits; i++) {
123 max_value *= 10;
124 }
125
126 if (data >= max_value) {
127 // Si el número es demasiado grande para el número de dígitos
128 return -1; // Error
129 }
130
131 // Convertimos el número a BCD y almacenamos cada dígito en el arreglo
132 for (uint8_t i = 0; i < digits; i++) {
133 bcd_number[digits - 1 - i] = data % 10; // Almacenamos el dígito menos significativo
134 data /= 10; // Quitamos ese dígito del número
135 }
136
137 return 0; // Éxito
138}

141void BCDtoGPIO(uint8_t digit, gpioConfig_t *gpio_config) {
142 for (uint8_t i = 0; i < N_BITS; i++) {
143 GPIOInit(gpio_config[i].pin, gpio_config[i].dir);
144 }
145
146 for (uint8_t i = 0; i < N_BITS; i++) {
147 if ((digit & (1 << i)) == 0) {
148 GPIOOff(gpio_config[i].pin);
149 } else {
150 GPIOOn(gpio_config[i].pin);
151 }
152 }
153}
154// Función para mostrar un número en el display LCD
155void displayNumberOnLCD(uint32_t data, gpioConfig_t *data_gpio_config, gpioConfig_t *digit_gpio_config) {
156 uint8_t bcd_array[LCD_DIGITS];
157
158 // Convertir el número a formato BCD
159 if (convertToBcdArray(data, LCD_DIGITS, bcd_array) != 0) {
160 printf("Error: el número es demasiado grande para la cantidad de dígitos.\n");
161 return;
162 }
163
164 // Inicializar los pines del LCD
165 for (uint8_t i = 0; i < LCD_DIGITS; i++) {
166 GPIOInit(digit_gpio_config[i].pin, digit_gpio_config[i].dir);
167 }
168
169 // Mostrar cada dígito en el display
170 for (uint8_t i = 0; i < LCD_DIGITS; i++) {
171 // Apagar todos los dígitos
172 for (uint8_t j = 0; j < LCD_DIGITS; j++) {
173 GPIOOff(digit_gpio_config[j].pin);
174 }
175
176 // Configurar el dígito actual para que esté encendido
177 GPIOOn(digit_gpio_config[i].pin);
178
179 // Mostrar el valor BCD del dígito
180 BCDtoGPIO(bcd_array[i], data_gpio_config);
